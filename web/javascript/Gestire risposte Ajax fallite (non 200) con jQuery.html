jQuery permette di settare il comportamento di ogni chiamata ajax effettuata con il framework in questo modo:

$.ajaxSetup({complete: onRequestCompleted});

function onRequestCompleted(xhr,textStatus) {
  if (xhr.status == 302) {
    location.href = xhr.getResponseHeader("Location");
  }
}
