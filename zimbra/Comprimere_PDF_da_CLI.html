Script per la conversione che vuole come parametri in ingresso la directory sorgente e la directory di destinazione

#!/bin/bash
for i in $(ls $1)
do
    echo converting $1/$i in $2/`basename $i`
    gs -dBATCH -dNOPAUSE -q -sDEVICE=pdfwrite -dPDFSETTINGS=/ebook -sOutputFile=$2/`basename $i` $1/$i
done
