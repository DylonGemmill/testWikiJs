Se siamo su Ubuntu 12.04 o inferiori è possibile installare un backport da neuro.debian.net che consente l'unban dinamico da linea di comando. Per la 12.04:

wget -O- http://neuro.debian.net/lists/precise.de-m.full |  tee /etc/apt/sources.list.d/neurodebian.sources.list
apt-key adv --recv-keys --keyserver hkp://pgp.mit.edu:80 0xA5D32F012649A5A9
apt-get update
apt-get install fail2ban

(se si installa questa versione il filtro sasl si chiama postfix-sasl)

 

il file principale di gestione è jail.conf, ma non si modifica direttamente. Invece creare il file /etc/fail2ban/jail.local con il seguente contenuto, modificando opportunamente i parametri ignoreip, bantime, maxretry, destemail:

[DEFAULT]

# elenco di ip, netmask per cui il ban non ha effetto. di solito si inserisce la rete locale, o degli ip fissi fidati
ignoreip = 127.0.0.1 10.22.22.0/24
bantime  = 1200
maxretry = 3

# "backend" specifies the backend used to get files modification. Available
# options are "gamin", "polling" and "auto".
# yoh: For some reason Debian shipped python-gamin didn't work as expected
#      This issue left ToDo, so polling is default backend for now
backend = polling

destemail = admin@yetopen.it

# Default action to take: ban only
action = %(banaction)s[name=%(__name__)s, port="%(port)s", protocol="%(protocol)s]


[ssh]

enabled = true
port    = ssh
filter  = sshd
logpath  = /var/log/auth.log
maxretry = 5


[apache]

enabled = true
port    = http
filter  = apache-auth
logpath = /var/log/apache*/*error.log
maxretry = 5


[apache-noscript]

enabled = false
port    = http
filter  = apache-noscript
logpath = /var/log/apache*/*error.log
maxretry = 5


[vsftpd]

enabled  = false
port     = ftp
filter   = vsftpd
logpath  = /var/log/auth.log
maxretry = 5


[proftpd]

enabled  = false
port     = ftp
filter   = proftpd
logpath  = /var/log/auth.log
failregex = proftpd: \(pam_unix\) authentication failure; .* rhost=<HOST>
maxretry = 5


[wuftpd]

enabled  = false
port     = ftp
filter   = wuftpd
logpath  = /var/log/auth.log
maxretry = 5


[postfix]

enabled  = false
port     = smtp
filter   = postfix
logpath  = /var/log/mail.log
maxretry = 5


[sasl]

enabled  = true
port     = smtp
filter   = sasl 
logpath  = /var/log/mail.log
maxretry = 5

Visti gli innumerevoli tentativi di sfondare phpmyadmin in alcuni casi è opportuno anche bloccare gli host che effettuano test alla cieca sulle pagine del server, visibili dagli errori 404 (file not found) in access.log. Per questo in jail.local aggiungere:

[apache-404block]
enabled = true
port = http
filter = apache-404block
logpath = /var/log/apache2/*access.log
bantime = 180
maxretry = 5

modificando in base alle proprie esigenze port e bantime. Quindi creare il file /etc/fail2ban/filter.d/apache-404block.conf con il seguente contenuto

[Definition]
failregex = ^<HOST> - - \[.*\] "GET /.* HTTP/1\.[01]" 404 [0-9]+.*$

ignoreregex =

Per mettere sotto controllo anche asterisk, aggiungere la seguente sezione al file jail.conf

 

[asteriskiptables]
enabled  = true
filter   = asterisk
action   = iptables-allports[name=ASTERISK, protocol=all]
logpath  = /var/log/asterisk/messages
maxretry = 5
bantime = 4000

 

e creare il file /etc/fail2ban/filter.d/asterisk.conf con il seguente contenuto

 

# Fail2Ban configuration file
#
#
# $Revision: 250 $
#

[INCLUDES]

# Read common prefixes. If any customizations available -- read them from
# common.local
#before = common.conf


[Definition]

#_daemon = asterisk

# Option:  failregex

#          be used for standard IP/hostname matching and is only an alias for
#          (?:::f{4,6}:)?(?P<host>\S+)
# Values:  TEXT
#

failregex = NOTICE.* .*: Registration from '.*' failed for '<HOST>' - Wrong password
            NOTICE.* .*: Registration from '.*' failed for '<HOST>' - No matching peer found
            NOTICE.* .*: Registration from '.*' failed for '<HOST>' - Username/auth name mismatch
            NOTICE.* .*: Registration from '.*' failed for '<HOST>' - Device does not match ACL
            NOTICE.* .*: Registration from '.*' failed for '<HOST>' - Peer is not supposed to register
            NOTICE.* <HOST> failed to authenticate as '.*'$
            NOTICE.* .*: No registration for peer '.*' \(from <HOST>\)
            NOTICE.* .*: Host <HOST> failed MD5 authentication for '.*' (.*)
            NOTICE.* .*: Failed to authenticate user .*@<HOST>.*

Riavviare fail2ban.

Se in fail2ban.log appare un errore 400 sulla riga con iptables e multiport, assicurarsi che il modulo xt_multiport sia caricato.

 

Per lo sblocco manuale di un IP:

fail2ban-client set pureftpd unbanip 188.186.9.132

dove pureftpd è il nome del filtro. Per vedere l'elenco dei filtri:

fail2ban-client status
