Scaricare Java in /tmp

tar xvzf /tmp/jre-7u21-linux-x64.tar.gz 
update-alternatives --install /usr/bin/java java /opt/java/jre1.7.0_21/bin/java 1
update-alternatives --set java /opt/java/jre1.7.0_21/bin/java
ln -sf /opt/java/jre1.7.0_21/lib/amd64/libnpjp2.so /usr/lib/mozilla/plugins/
